\frame{
    \frametitle{Project Tracing}
    \begin{block}{Ziel}
        Das Ziel von Project Tracing ist die Verfolgung bestimmter
        Projektparameter: Wo steht mein Projekt jetzt in diesem
        Moment?
    \end{block}
    
    \begin{itemize}
        \item BWL-Sicht: Zeit, Resourcen, Kosten, Umfang
        \item Entwickler-Sicht: Qualität, Velocity, gute Dokumentation, Source-Metriken...
    \end{itemize}
}
