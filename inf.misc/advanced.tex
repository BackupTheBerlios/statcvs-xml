\section{Tipps}

\subsection{Command Line}
\frame[all:1]{
  \frametitle{Command Line Optionen}
  
  Eine übersicht der wichtigsten Parameter:

  \begin{itemize}
  \item {\bf -g}
    Eine Liste aller Goals anzeigen.
  \item {\bf -o} 
    Offline Modus. Nur Goals ausführen, die keine
    Internet Verbindung benötigen.
  \item {\bf -p projekt-datei} 
    Notwendig für Aufrufe aus
    Nicht-Projekt-Verzeichnissen (z.B. in einem Cron Job).
  \end{itemize}
}

\subsection{Konfiguration}
\frame[all:1]{
  \frametitle{Konfiguration}
  
  Persönliche Einstellungen können in \$HOME/build.properties
  vorgenommen werden.

  \begin{block}{\$HOME/build.properties}
    \begin{alltt}
maven.repo.local=/usr/local/maven-1.0-rc1/repository
maven.home.local=/usr/local/maven-1.0-rc1
java.compiler=modern
maven.username=nick\end{alltt}
  \end{block}
}

\section{Customizing}

\subsection{Customizing}
\frame[all:1]{
  \frametitle{Customizing}
  
  \begin{itemize}
  \item Jelly: Code und Layout bunt durcheinander gemischt
  \item xnap.jsl Ein Beispiel für ein 3 Column Layout
  \item Includieren von php Skripten 
  \end{itemize}

  \begin{example}
    \begin{alltt}
<jsl:template match="include" trim="false">
 <x:set var="_file" select="string(@file)"/>
 <jsl:comment>#include virtual="${_file}"</jsl:comment>
</jsl:template>\end{alltt}
  \end{example}

  \begin{example}
    \begin{alltt}
<section name="FAQ">
<p><include file="faq.php"/></p>\end{alltt}
  \end{example}
}
