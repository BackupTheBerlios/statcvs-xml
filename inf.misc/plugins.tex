\subsection{Metriken}
\frame{
  \frametitle{Style Guide Checking}

  {\bf CheckStyle Plugin}
  \begin{itemize}
  \item Prüft den Source Code anhand von Regeln
  \item Bericht enthält Links auf Fehlerzeile im XRef
  \item Konfiguration über eine Properties Datei
  \item Mit Maven 1.0-rc1 ausgelieferte Version benutzt veraltetes
  CheckStyle 2.1 (CVS Version benutzt bereits 3.3)
  \end{itemize}
  
  \begin{block}{Konfiguration (project.properties)}
    \begin{alltt}maven.checkstyle.format = (sun|turbine|avalon)\\
maven.checkstyle.properties = mystyle.properties\end{alltt}
  \end{block}
}

\frame[all:1]{
  \frametitle{Style Guide Checking}

  {\bf Jalopy Plugin}
  \begin{itemize}
  \item Formatiert Source Code anhand von Regeln
  \item Transformiert unter anderem
    \begin{itemize}
    \item Geschweifte Klammern
    \item Leerzeichen und -zeilen
	\item Einrückungen
    \item Zeilenumbrüche
    \end{itemize}
  \item Kann JavaDoc Kommentare, Kopf- und Fusszeilen generieren
  \end{itemize}

  \begin{block}{Konfiguration (project.properties)}
    \begin{alltt}maven.jalopy.style = mystyle.xml\end{alltt}
  \end{block}
}

\frame[all:1]{
  \frametitle{Source Code Metriken}
  
  {\bf PMD Plugin}
  \begin{itemize}
  \item Prüft den Source Code anhand von Regeln auf potientielle Fehler
  \item Findet unter anderem
    \begin{itemize}
    \item Unbenutzte Variablen
    \item Leere catch-Blöcke
    \item Leere if-Anweisungen, 
    \item Duplizierte import Anweisungen
    \item Nicht benutzte private Methoden
	\item Cyclomatic Complexity
    \end{itemize}
  \item Anpassung über vordefinierte Regeldateien
  \end{itemize}
  
  \begin{block}{Konfiguration (project.properties)}
    \begin{alltt}maven.pmd.rulesetfiles
  = \${plugin.resources}/rulesets/strings.xml,
    \${plugin.resources}/rulesets/junit.xml, ...\end{alltt}
  \end{block}
}

\frame[all:1]{
  \frametitle{Source Code Metriken}  

  {\bf JDepend Plugin}
  \begin{itemize}
  \item Prüft die Entwurfs Qualität von Source Code anhand von Regeln
  \item Misst die 
    \begin{itemize}
    \item Erweiterbarkeit
    \item Wiederverwenbarkeit
    \item Wartbarkeit
    \end{itemize}
  \item Zeigt Metriken pro Packet an
  \end{itemize}

  \vspace{0.5cm}
 
  \emph{Je nach Anzahl der Pakete und Fehler kann der Berichte sehr
    gross werden (z.B. XNap mit ca. 65.000 SLOC: 742 kb).}
}

\subsection{Test}
\frame[all:1]{
  \frametitle{Test und Überdeckung}

  {\bf JUnit Report Plugin}
  \begin{itemize}
  \item Erstellt einen Bericht über die Ausführung der Testfälle
  \end{itemize}

  \vspace{0.5cm}
  
  {\bf Clover Plugin}
  \begin{itemize}
  \item Misst die Testüberdeckung
  \item Automatische Instrumentierung des Source Codes
  \item Messung der Anweisungs- und Zweigüberdeckung
  \item HTML Ausgabe im Frameset ähnlich JavaDoc
  \item Frameset nicht in die Projekt Seite integriert
  \item Clover ist nicht frei
  \item Kostenlose Lizenz für Open-Source Projekte auf Anfrage
  \end{itemize}

  \begin{block}{Konfiguration (project.properties)}
    \begin{alltt}maven.junit.fork = true\end{alltt}
  \end{block}
}

\frame[all:1]{
  \frametitle{Testüberdeckung}
  
  {\bf JCoverage Plugin}
  \begin{itemize}
  \item Funktionalität äquivalent zu Clover
  \item GPL Version verfügbar
  \item Unbrauchbare Standardkonfiguration
  \item Fehler im aktuellen Maven 1.0-rc1 Release
  \end{itemize}

  \begin{block}{Konfiguration (project.properties)}
    \begin{alltt}maven.jcoverage.report.template = jcoverage\end{alltt}
  \end{block}

  \vspace{0.5cm}
  
  {\bf GCover Plugin}
  \begin{itemize}
  \item Formatiert die Ausgabe von gcov als HTML Frameset
  \item gcov kann Testüberdeckung für gcc Kompilate auswerten
  \item Bisher nicht in Maven integriert
  \end{itemize}
}

\section{SCM Plugins}
\subsection{SCM Plugins}
\frame[all:1]{
  \frametitle{Commit Log Plugins}

  {\bf ChangeLog Plugin}
  \begin{itemize}
  \item Zeigt den SCM Commit Log als Bericht an
  \item Fügt Links auf die jeweiligen Revisionen in ViewCVS ein
  \item Unterstützt mehrere Soure Control Management Werkzeuge
    \begin{itemize}
    \item CVS
    \item Perforce
    \item StarTeam
    \item IBM Rational ClearCase 
    \item Subversion
    \end{itemize}
  \end{itemize}
}

\subsection{CVS Plugins}
\frame[all:1]{
  \frametitle{CVS Plugins}

  {\bf Developer Activity Plugin}
  \begin{itemize}
  \item Zeigt zu jedem Entwickler
    \begin{itemize}
    \item Anzahl Commits
    \item Anzahl der geänderten Dateien
    \end{itemize}
  \item Unterstützt nur CVS
  \end{itemize}

  \vspace{0.5cm}

  {\bf File Activity Plugin}
  \begin{itemize}
  \item Zeigt zu jeder Datei die Anzahl der Änderungen
  \item Unterstützt nur CVS
  \end{itemize}
}

\frame[all:1]{
  \frametitle{CVS Plugins}

  {\bf StatCvs Plugin}
  \begin{itemize}
  \item Erzeugt Metriken aus dem CVS Log
    \begin{itemize}
    \item Anzahl der Lines of Code pro Entwickler
    \item Commit Log (ähnlich dem ChangeLog Plugin)
    \item Anzahl der Dateien
    \item Grösse der Dateien
    \item Grösse der Module
    \item Aktivität in den Modulen
    \end{itemize}
  \item Graphische Darstellung mit Charts
  \item Basiert auf StatCvs-XML
  \item Unterstützt nur CVS
  \end{itemize}
}

\subsection{Standard Plugins}
\frame[all:1]{
  \frametitle{Standard Plugins}

  {\bf License Plugin}
  \begin{itemize}
  \item Zeigt die Lizenz an
  \end{itemize}

  \begin{block}{Konfiguration (project.properties)}
    \begin{alltt}maven.license.licenseFile = COPYING\end{alltt}
  \end{block}

  \vspace{0.5cm}

  {\bf Changes Plugin}
  \begin{itemize}
  \item Zeigt eine Versions Historie an
  \item Konfiguration über eine XML Datei
  \end{itemize}

  \begin{Beispiel}
\begin{alltt}\scriptsize
xsltproc changes.xsl xdocs/changes.xml | fold -s 
  | perl -n -i -e 'print "    " if $_ =~ /^[A-Za-z]/
                   && $_ !~ /^Version/;' -e 'print $_' > CHANGES
\end{alltt}
    \end{Beispiel}
}
